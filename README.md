HTML - Hyper Text Markup Languege
HTML ----> CONTEÚDO
<destacado>TEXTO</destacado>
<comando></comando>
<comando> --> <tag>
<p></p> --> parágrafo
<br> --> quabra linha
<img src-"foto.jpg"> --> imagem
<a href="https://sp.senac.br">Senac</a> --> link
